### 五编写 code 段中的代码，将 a段 和 b段 数据依次相加，结果存入c段

```assembly
assume cs:code
a segment
      db 1,2,3,4,5,6,7,8
a ends
b segment
      db 1,2,3,4,5,6,7,8
b ends
c segment
      db 0,0,0,0,0,0,0,0
c ends
code segment
    start:
          mov  cx,8
          mov  bx,0
    s:    mov  dx,0
          mov  ax,a
          mov  ds,ax
          mov  dl,ds:[bx]
          mov  ax,b
          mov  ds,ax
          add  dl,ds:[bx]
          mov  ax,c
          mov  ds,ax
          mov  ds:[bx],dl
          add  bx,1
          loop s
          mov  ax,4c00H
          int  21H
code ends
end start
```
